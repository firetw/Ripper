                $j('#postalCodeChoiceLayer').show();
            }
            $j('#city').removeClass("error");
            $j('#postalCode').addClass("valid");
        }
        if ( jsonData.postalCode.city ) {
            $j('#city').val(jsonData.postalCode.city);
            $j('#city').removeClass("error");
            $j('#postalCode').addClass("valid");
        }
        if ( jsonData.postalCode.province ) {
            $j('#province').val(jsonData.postalCode.province);
            $j('#province').removeClass("error");
            $j('#postalCode').addClass("valid");
        }
        //if ( $j("label[for='postalCode']") ) {
        //    $j("label[for='postalCode']").remove();
        //}
        if ( $j('label[for=postalCode][class=error]').length > 0 ) {
			$j('label[for=postalCode][class=error]').remove();
		}
    } else if ( jsonData.postalCode_chkMsg ) {
        
        $j('#postalCode').addClass("error");
        //if ( $j("label[for='postalCode']") ) {
        //	$j("label[for='postalCode']").remove();
        //}
        if ( $j('label[for=postalCode][class=error]').length > 0 ) {
			$j('label[for=postalCode][class=error]').remove();
		}
        $j("<label for='postalCode' generated='true' class='error'>"+jsonData.postalCode_chkMsg+"</label>").insertAfter('#postalCode');
        
        $j('#city').val('');
        $j('#city').addClass("error");
        $j('#province').val('');
        $j('#province').addClass("error");
        chkResult = false;
    }
    return chkResult;
}
